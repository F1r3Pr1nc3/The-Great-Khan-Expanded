#########################################
#
# on_action_events patch rebuild outpost for orphaned colonies
#
#########################################
namespace = action
event = {
	id = action.90
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		any_playable_country = {
			any_owned_planet = {
				is_controlled_by = prev
				solar_system = {
					NOT = { exists = starbase }
				}
			}
		}
	}
	immediate = {
		every_system = {
			limit = {
				NOT = { exists = starbase }
				any_system_colony = {
					is_colony = yes
					has_ground_combat = no
					owner = {
						OR = {
							merg_is_standard_empire = yes
							is_country_type = awakened_marauders
						}
					}
					solar_system = {
						NOR = {
							any_fleet_in_system = {
								exists = owner
								owner = {
									NOT = { is_same_empire = prevprevprev.owner }
									OR = {
										is_hostile = prevprevprev.owner
										is_at_war_with = prevprevprev.owner
									}
								}
							}
							any_system_colony = {
								NOT = { is_same_value = prevprev }
								owner = {
									NOT = { is_same_empire = prevprevprev.owner }
									OR = {
										merg_is_standard_empire = yes
										is_country_type = awakened_marauders
									}
								}
							}
						}
					}
				}
			}
			random_system_colony = {
				limit = {
					exists = owner
					has_ground_combat = no
					owner = {
						OR = {
							merg_is_standard_empire = yes
							is_country_type = awakened_marauders
						}
					}
					solar_system = {
						NOR = {
							any_fleet_in_system = {
								exists = owner
								owner = {
									NOT = { is_same_empire = prevprevprev.owner }
									OR = {
										is_hostile = prevprevprev.owner
										is_at_war_with = prevprevprev.owner
									}
								}
							}
							any_system_colony = {
								owner = {
									NOT = { is_same_empire = prevprev.owner }
									OR = {
										merg_is_standard_empire = yes
										is_country_type = awakened_marauders
									}
								}
							}
						}
					}
				}
				planet_event = { id = action.91 }
			}
		}
	}
}

planet_event = {
	id = action.91
	title = action.91.name
	desc = action.91.desc
	picture = GFX_evt_star_chart
	show_sound = event_default
	location = root
	is_triggered_only = yes
	immediate = {
		solar_system = {
			if = {
				limit = {
					root.owner = { is_country_type = awakened_marauders }
				}
				create_starbase = {
					size = starbase_marauder
					owner = root.owner
					effect = {
						fleet = {
							every_owned_ship = {
								reduce_hp_percent = 98 set_disabled = no
							}
						}
					}
				}
			} else = {
				create_starbase = {
					size = starbase_outpost
					owner = root.owner
				}
				if = {
					limit = { root.owner = { is_specialist_subject_type = { TYPE = bulwark } } }
					starbase = {
						add_starbase_component = {
							component = "STARBASE_AURA_BULWARK_ARRAY"
						}
					}
				}
			}
		}
	}
	option = {
		name = OK
		tooltip = {
			solar_system = {
				create_starbase = {
					size = starbase_outpost
					owner = root.owner
				}
			}
		}
	}
}
